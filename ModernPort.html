def simulate_portfolio_optimization(hist_data, num_simulations):
    # Remove tickers with all NaN values from hist_data and expected_returns
    hist_data = hist_data.loc[:, hist_data.notna().any()]
    expected_returns = np.nanmean(hist_data.pct_change(), axis=0)
    expected_returns = pd.Series(expected_returns[~np.isnan(expected_returns)], index=hist_data.columns[~np.isnan(expected_returns)])

    # Calculate the covariance matrix, removing rows and columns with all NaN values
    cov_matrix = hist_data.pct_change().cov()
    cov_matrix = cov_matrix.loc[cov_matrix.index.notna(), cov_matrix.columns.notna()]

    # Define the constraints for the optimization problem
    constraints = ({'type': 'eq', 'fun': lambda x: np.sum(x) - 1})

    # Set the bounds for the optimization problem
    bounds = tuple((0, 1) for _ in range(len(hist_data.columns)))

    # Initialize variables to hold the best portfolio metrics
    best_portfolio_return = -np.inf
    best_portfolio_var = np.inf
    best_portfolio_volatility = np.inf
    best_weights = None

    # Simulate the portfolio optimization process
    for i in range(num_simulations):
        print(i)
        # Generate random weights
        weights = np.random.rand(len(expected_returns))
        weights /= np.sum(weights)

        # Calculate the portfolio return
        portfolio_return = np.sum(expected_returns * weights)

        # Calculate the portfolio variance
        portfolio_var = np.dot(weights.T, np.dot(cov_matrix, weights))

        # Calculate the portfolio standard deviation (volatility)
        portfolio_volatility = np.sqrt(portfolio_var)

        # Check if this is the best portfolio so far
        if portfolio_return > best_portfolio_return:
            best_portfolio_return = portfolio_return
            best_portfolio_var = portfolio_var
            best_portfolio_volatility = portfolio_volatility
            best_weights = weights

    # Create a Pandas Series with the optimal weights and tickers as the index
    optimal_weights = pd.Series(best_weights, index=hist_data.columns)

    # Return the best portfolio metrics as a dictionary
    return {'Expected Returns': expected_returns,
            'Covariance Matrix': cov_matrix,
            'Optimal Weights': optimal_weights,
            'Portfolio Return': best_portfolio_return,
            'Portfolio Variance': best_portfolio_var,

from scipy.optimize import minimize

def calculate_portfolio_metrics(hist_data):
    # Remove tickers with all NaN values from hist_data and expected_returns
    hist_data = hist_data.loc[:, hist_data.notna().any()]
    expected_returns = np.nanmean(hist_data.pct_change(), axis=0)
    expected_returns = pd.Series(expected_returns[~np.isnan(expected_returns)], index=hist_data.columns[~np.isnan(expected_returns)])

    # Calculate the covariance matrix, removing rows and columns with all NaN values
    cov_matrix = hist_data.pct_change().cov()
    cov_matrix = cov_matrix.loc[cov_matrix.index.notna(), cov_matrix.columns.notna()]

    # Define the constraints for the optimization problem

    # Define the objective function for the optimization problem
    # Define the objective function for the optimization problem
    def objective_function(weights):
        portfolio_var = np.dot(weights.T, np.dot(cov_matrix, weights))
        return portfolio_var


    # Define the constraints for the optimization problem
    constraints = ({'type': 'eq', 'fun': lambda x: np.sum(x) - 1})

    # Set the bounds for the optimization problem
    bounds = tuple((0, 1) for _ in range(len(hist_data.columns)))

    # Set an initial guess for the optimization problem
    initial_guess = np.array([1.0 / len(hist_data.columns)] * len(hist_data.columns))

    # Run the optimization problem to find the optimal weights
    result = minimize(objective_function, initial_guess, method='SLSQP', bounds=bounds, constraints=constraints)

    # Get the optimal weights
    weights = result.x

    # Calculate the portfolio return
    portfolio_return = np.sum(expected_returns * weights)

    # Calculate the portfolio variance
    portfolio_var = np.dot(weights.T, np.dot(cov_matrix, weights))

    # Calculate the portfolio standard deviation (volatility)
    portfolio_volatility = np.sqrt(portfolio_var)

    # Return the portfolio metrics as a dictionary
    return {'Expected Returns': expected_returns,
