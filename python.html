# we write a function in order to prepare the text data by removing and cleaning up a few things
def data_preparation(reddit):
    try:
        # Step 1: Remove brackets incse they do end up existing
        reddit = re.sub(r'[][)(]', ' ',reddit)

        # Step 2 : make everything lowercase so it can be understood the same way
        reddit = reddit.lower()

        #Step 3: Create tokens for the given words
        tokens = nltk.word_tokenize(reddit)

        # get rid of all useless connector words that we arent worried about
        reddit = [word for word in tokens if word not in stop_words]

        # now we are lemmenization so bascally we are reducing certain words into their base
        reddit = [lemmatizer.lemmatize(word) for word in reddit]
        reddit = ' '.join(reddit)

        # get rid of reddit 'r/' from any parts of the text
        reddit = re.sub(r'r/', '', reddit)

        reddit =  re.sub(r',', ' ', reddit)

        return reddit
    except Exception as ex:
        print('Line:', reddit,'\n\n')
        print('Exception: ', ex,'\n')


reddit_data['ProcessedTitles'] = reddit_data['Title'].apply(lambda Title: data_preparation(Title))

def evaluate_model(model, X_train, y_train, X_test, y_test):
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)
    accuracy = accuracy_score(y_test, y_pred)
    return accuracy
